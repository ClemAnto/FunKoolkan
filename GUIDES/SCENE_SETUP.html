3. Modifiche al codice necessarie

Importante: GameManager.ts crea attualmente GameLayer, UILayer e tutti i nodi HUD via codice. Con la scena pre-strutturata deve trovare i nodi esistenti invece di crearne di nuovi.

GameLayer e UILayer — trova invece di creare

// GameManager.start() — sostituisci le righe di new Node()
this.gameLayer = this.node.parent!.getChildByName('GameLayer')
  ?? (() => { const n = new Node('GameLayer'); n.setParent(this.node.parent!); return n; })();

this.uiLayer = this.node.parent!.getChildByName('UILayer')
  ?? (() => { const n = new Node('UILayer'); n.setParent(this.node.parent!); return n; })();

createHud() — legge nodi pre-esistenti

// Trova HUD pre-esistente (già in scena) o lo crea come fallback
const hud = this.uiLayer.getChildByName('HUD') ?? new Node('HUD');
if (!hud.parent) hud.setParent(this.uiLayer);

// Legge i Label pre-esistenti invece di crearli
this.scoreLabel = hud.getChildByName('ScoreSec')
  ?.getChildByName('ScoreValue')?.getComponent(Label) ?? null;
this.roundLabel = hud.getChildByName('RoundSec')
  ?.getChildByName('RoundValue')?.getComponent(Label) ?? null;
this.mergesLabel = hud.getChildByName('MergesSec')
  ?.getChildByName('MergesValue')?.getComponent(Label) ?? null;
this.nextPreviewNode = hud.getChildByName('NextSec')
  ?.getChildByName('NextPreview') ?? null;
this.timerLabel = hud.getChildByName('TimerSec')
  ?.getChildByName('TimerValue')?.getComponent(Label) ?? null;

GameOverLine e LauncherAnchor — riposizionati da Track.ts

// In Track.ts — relayout() riposiziona i nodi anchor
const goLine = this.node.getChildByName('GameOverLine');
if (goLine) goLine.setPosition(0, GAME_OVER_LINE_Y, 0);

const launcher = this.node.getChildByName('LauncherAnchor');
if (launcher) launcher.setPosition(0, TRACK_TOP_Y, 0);
